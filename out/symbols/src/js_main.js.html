 12 /**
 13  * Receives up to 3 parameters and returns a newly-created HTML element
 14  *
 15  * @function createElemWithText
 16  * @param {String} htmlElem - The HTML element string name to be created (h1, p, button, etc) (optional)
 17  * @param {String} textContent - The text content of the element to be created (optional)
 18  * @param {String} className - A className if one is to be applied (optional)
 19  * @returns {HTMLElement} The created element.
 20  */
 21 const createElemWithText = (htmlElem = "p", textContent = "", className) => {
 22   // Create the new element
 23   const newElement = document.createElement(htmlElem);
 24 
 25   // Set the other attributes
 26   newElement.textContent = textContent;
 27   if (className) newElement.className = className;
 28 
 29   // Return the created element
 30   return newElement;
 31 };

 33 /**
 34  * Creates an array of select options representing all blog users
 35  *
 36  * @function createSelectOptions
 37  * @param {JSON} users - JSON data representing the users of the blog
 38  * @returns {any[]} An array of options HTMLElements representing blog users
 39  */
 40 const createSelectOptions = (users) => {
 41   // Return early if no parameter received
 42   if (!users) return undefined;
 43 
 44   // Create array of options elements
 45   const optionElements = [];
 46 
 47   // Loop through user data
 48   users.forEach((user) => {
 49     // Create a new option
 50     const option = document.createElement("option");
 51 
 52     // Add user data to the option
 53     option.value = user.id;
 54     option.textContent = user.name;
 55 
 56     // Push new option to array
 57     optionElements.push(option);
 58   });
 59 
 60   // Return array of options elements
 61   return optionElements;
 62 };

 64 /**
 65  * Toggles the class 'hide' on the section element containing the comments of the specified post

109 /**
110  * Deletes all children of a specified HTML parent element
111  *
112  * @function deleteChildElements
113  * @param {Element} parentElement - The element from which to remove all children
114  * @returns {Element} The parent element
115  */
116 const deleteChildElements = (parentElement) => {
117   const isDOM = (el) => el instanceof Element;
118 
119   // Return early if no element received. or if the parameter is not a valid element name
120   if (!parentElement || !isDOM(parentElement)) return undefined;
121 
122   // Define the child
123   let child = parentElement.lastElementChild;
124 
125   // While a child still exists, remove it
126   while (child) {
127     parentElement.removeChild(child);
128     child = parentElement.lastElementChild;
129   }
130 
131   // Return the parentElement once all children removed
132   return parentElement;
133 };

141 /**
142  * Adds event listeners to all buttons found in main, calling the toggleComments function
143  *
144  * @function addButtonListeners
145  * @returns {NodeListOf<Element>} All buttons nested inside the main element
146  * @requires toggleComments()
147  */
148 const addButtonListeners = () => {
149   // Select all buttons inside the main element
150   const buttons = document.querySelectorAll("main button");
151 
152   // Return if no buttons
153   if (!buttons) return;
154 
155   // Loop through the NodeList of buttons
156   buttons.forEach((button) => {
157     // Get the postId
158     const postId = button.dataset.postId;
159 
160     // Add click event listener
161     button.addEventListener("click", (event) => {
162       toggleComments(event, postId);
163     });
164   });
165 
166   return buttons;
167 };

169 /**
170  * Removes the button listeners from all buttons inside the main element
171  *
172  * @function removeButtonListeners
173  * @returns {NodeListOF<Element>} All buttons inside the main element
174  * @requires toggleComments()
175  */
176 const removeButtonListeners = () => {
177   // Select all buttons inside the main element
178   const buttons = document.querySelectorAll("main button");
179 
180   // Return if no buttons
181   if (!buttons) return;
182 
183   // Loop through the NodeList of buttons
184   buttons.forEach((button) => {
185     // Get the postId
186     const postId = button.dataset.id;
187 
188     // Remove the event listener
189     button.removeEventListener("click", (event) => {
190       toggleComments(event, postId);
191     });
192   });
193 
194   return buttons;
195 };

197 /**
198  * Creates a document fragment containing all the comments from the JSON argument
199  *
200  * @function createComments
201  * @param {JSON} commentData - JSON comments data
202  * @returns {DocumentFragment} The generated fragment with all comments
203  * @requires createElemWithText()
204  */
205 const createComments = (commentData) => {
206   // Return undefined if no parameter given
207   if (!commentData) return undefined;
208 
209   // Create a fragment element
210   const frag = document.createDocumentFragment();
211 
212   // Loop through the comments
213   commentData.forEach((comment) => {
214     // Create the article
215     const article = document.createElement("article");
216 
217     // Create the article elements and append to the article
218     const h3 = createElemWithText("h3", comment.name);
219     const body = createElemWithText("p", comment.body);
220     const from = createElemWithText("p", `From: ${comment.email}`);
221     article.append(h3, body, from);
222 
223     // Append the article to the fragment
224     frag.append(article);
225   });
226 
227   // Return the fragment element
228   return frag;
229 };

324     if (found.length === 0) throw new Error("User not found");

326   } catch (error) {
327     console.log(error);
328   }

360 /**
361  * Creates a section element containing all comments for a specified post
362  *
363  * @async
364  * @function displayComments
365  * @param {String} postId - The post for which to display comments
366  * @returns {Promise<HTMLElement>} The Section element containing the post comments
367  * @requires getPostComments()
368  * @requires createComments()
369  */
370 const displayComments = async (postId) => {
371   if (!postId) return undefined;
372 
373   const section = document.createElement("section");
374   section.dataset.postId = postId;
375   section.classList.add("comments", "hide");
376 
377   const comments = await getPostComments(postId);
378   const fragment = createComments(comments);
379 
380   section.appendChild(fragment);
381   return section;

384 /**
385  * Creates a document fragment containing all posts within the specified JSON data
386  *
387  * @async
388  * @function createPosts
389  * @param {JSON} postsJSON - JSON data representing posts to create
390  * @returns {DocumentFragment} The HTML for all posts within the given JSON data
391  * @requires createElemWithText()
392  * @requires getUser()
393  * @requires displayComments()
394  */
395 const createPosts = async (postsJSON) => {
396   // Return early if undefined
397   if (!postsJSON) return undefined;
398 
399   // Create the document fragment to contain all posts
400   const fragment = document.createDocumentFragment();
401 
402   // Loop over each post
403   for (let post of postsJSON) {
404     // Create the post article element
405     const article = document.createElement("article");
406 
407     // Create all post elements
408     const h2 = createElemWithText("h2", post.title);
409     const p_body = createElemWithText("p", post.body);
410     const p_postId = createElemWithText("p", `Post ID: ${post.id}`);
411 
412     const author = await getUser(post.userId);
413     const p_author = createElemWithText(
414       "p",
415       `Author: ${author.name} with ${author.company.name}`
416     );
417     const p_catchphrase = createElemWithText(
418       "p",
419       `${author.company.catchPhrase}`
420     );
421 
422     const button = createElemWithText("button", "Show Comments");
423     button.dataset.postId = post.id;
424 
425     const section = await displayComments(post.id);
426 
427     // Append all post elements to the post article
428     article.append(
429       h2,
430       p_body,
431       p_postId,
432       p_author,
433       p_catchphrase,
434       button,
435       section
436     );
437 
438     // Append the article to the fragment containing all posts
439     fragment.appendChild(article);
440   }
441 
442   // Return the fragment containing all the posts
443   return fragment;
444 };

446 /**
447  * Creates posts from given JSON data and appends all to the document
448  *
449  * @async
450  * @function displayPosts
451  * @param {JSON} postsJSON - JSON data representing posts to create
452  * @returns {HTMLElement | DocumentFragment} The posts created
453  * @requires createPosts()
454  * @requires createElemWithText()
455  */
456 const displayPosts = async (postsJSON) => {
457   // Select the main element of the document
458   const main = document.querySelector("main");
459 
460   // Create the element for the posts
461   let element;
462   // If posts data was provided, create the posts from the JSON
463   if (postsJSON) {
464     element = await createPosts(postsJSON);
465   }
466   // Otherwise, create a default post
467   else {
468     element = createElemWithText(
469       "p",
470       "Select an Employee to display their posts."
471     );
472     element.classList.add("default-text");
473   }
474   // Append the posts to the document main element
475   main.append(element);
476   // Return the element appended to the document
477   return element;
478 };

486 /**
487  * An event listener, toggles the comments for a specific post
488  *
489  * @function toggleComments
490  * @param {Event} event - The event from a click event listener
491  * @param {String} postId - The post for which to toggle comments
492  * @returns {Element[]} An array containing the comment section element and the button element for the comments section
493  * @requires toggleCommentSection()
494  * @requires toggleCommentButton()
495  */
496 const toggleComments = (event, postId) => {
497   // Ensure that both the event and post id were provided
498   if (!event || !postId) return undefined;
499 
500   // Toggle comments section and button
501   event.target.listener = true;
502   const section = toggleCommentSection(postId);
503   const button = toggleCommentButton(postId);
504 
505   // Return the comment section and button
506   return [section, button];
507 };

509 /**
510  * Removes all posts from the page and adds all posts passed in as JSON
511  * 
512  * @async
513  * @function refreshPosts
514  * @param {JSON} postsJSON - posts JSON data
515  * @returns {any[]} An array of the results from the functions called: [removeButtons, main,
516 fragment, addButtons]
517  * @requires removeButtonListeners()
518  * @requires deleteChildElements()
519  * @requires displayPosts()
520  * @requires addButtonListeners()
521  */
522 const refreshPosts = async (postsJSON) => {
523   // Return early if json not provided
524   if (!postsJSON) return undefined;
525 
526   // Remove old posts
527   const removeButtons = removeButtonListeners();
528   let main = document.querySelector("main");
529   main = deleteChildElements(main);
530 
531   // Create new posts
532   const fragment = await displayPosts(postsJSON);
533   const addButtons = addButtonListeners();
534 
535   // Return the results of the functions called
536   return [removeButtons, main, fragment, addButtons];
537 };

539 /**
540  * Event handler for the select menu
541  *
542  * @async
543  * @function selectMenuChangeEventHandler
544  * @param {Event} event - The event generated
545  * @returns {any[]} An array with the userId, posts and the array returned from refreshPosts: [userId, posts, refreshPostsArray]
546  * @requires getUserPosts()
547  * @requires refreshPosts()
548  */
549 const selectMenuChangeEventHandler = async (event) => {
550   // User ID if contained in the event, or 1
551   const userId = event?.target?.value || 1;
552 
553   // get posts JSON data
554   const postsJSON = await getUserPosts(userId);
555 
556   // Refresh the posts
557   const refreshPostsArray = await refreshPosts(postsJSON);
558 
559   // Return an array with the userId, posts and the array returned from refreshPosts: [userId, posts, refreshPostsArray]
560   return [userId, postsJSON, refreshPostsArray];
561 };
